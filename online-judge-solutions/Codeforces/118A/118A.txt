#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
bool esVocal(char c){
	if(c=='a' || c=='e' || c=='i'|| c=='o'|| c=='u' || c=='y'){return true;}
	else{return false;}
}
int main(){
	string s,a;
	cin>>s;
	transform(s.begin(),s.end(),s.begin(),::tolower);
	for(int i=0;i<s.size();i++){
		if(!esVocal(s[i])){a+=s[i];}
	}
	for(int i=0;i<a.size();i++){
		cout<<"."<<a[i];
	}
	return 0;
}
