#include <iostream>
using namespace std;
int main(){
	int n,k,a[51],t=0,max;
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>a[i];
		if(i+1==k){max=a[i];}
	}
	for(int i=0;i<n;i++){
		if(a[i]>=max && a[i]>0){t++;}
	}
	cout<<t<<endl;
	return 0;
}
