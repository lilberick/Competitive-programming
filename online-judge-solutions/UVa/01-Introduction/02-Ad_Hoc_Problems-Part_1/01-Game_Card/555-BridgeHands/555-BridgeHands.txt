#include <stdio.h>
#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <algorithm>
using namespace std;
map<char,int> suit;
map<char,int> deck;
bool myfunction (string i,string j) {
	if(suit[i[0]] == suit[j[0]]){
		return (deck[i[1]] < deck[j[1]]);
	}else{
		return (suit[i[0]] < suit[j[0]]);
	}
}
int main(){
	suit['C'] = 0;
	suit['D'] = 1;
	suit['S'] = 2;
	suit['H'] = 3;
	
	deck['2'] = 2;

	deck['Q'] = 12;
	deck['K'] = 13;
	deck['A'] = 14;

	int i,j,current;
	char c;
	string dump;
	string cards,buffer;
	vector<string> players[4];
	vector<string>::iterator pl_it;
	map<char,int> dir;
	map<int,char> rdir;
	
	dir['S'] = 0;
	dir['W'] = 1;
	dir['N'] = 2;
	dir['E'] = 3;
	rdir[0] = 'S';
	rdir[1] = 'W';

	while(cin>>c){
