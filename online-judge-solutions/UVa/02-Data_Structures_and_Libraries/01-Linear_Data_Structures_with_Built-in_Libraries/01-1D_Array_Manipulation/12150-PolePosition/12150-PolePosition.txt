#include<iostream>
#include<set>
using namespace std;
int main(){
	int n;
	while(cin>>n,n){
		int c,p,carro[10005]={},bien=1;
		set<int> s;
		for(int i=0;i<n;i++){
			cin>>c>>p;
			if(i+p<0 || i+p>=n){bien=0;}
			if(bien){carro[i+p]=c;s.insert(i+p);}
		}
		if(s.size()!=n){bien=0;}
		if(bien==0){cout<<-1<<endl;}
		else{
			cout<<carro[0];
			for(int i=1;i<n;i++){cout<<" "<<carro[i];}
			cout<<endl;
		}
	}
	return 0;
}
